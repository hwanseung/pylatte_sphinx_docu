                length= 0

                param = dict( (k, v if len(v)>1 else v[0] ) for k, v in parse_qs(environ['wsgi.input'].read(length).decode(),True).items() )
            else:
                    param = dict()
        elif headerInfo["CONTENT_TYPE"].startswith("multipart/form-data"):
